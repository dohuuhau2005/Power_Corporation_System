const AES_SECRET_KEY = Buffer.from("DienLuc@2026_SecretKey_DienLuc@2", 'utf8'); 
const PRIVATE_KEY = `-----BEGIN PRIVATE KEY-----\n... (Nhét khóa của ông vào) ...\n-----END PRIVATE KEY-----`;

// 🔥 CÁCH CỦA BRO: CHƠI RAW SQL LUÔN, KHÔNG CẦN JSON
const rawSQLQuery = `INSERT INTO nhanvien (maNV, hoten, maCN) VALUES ('NV999', 'Huy', 'TP1')`;

// BƯỚC 1: Băm cái chuỗi SQL bằng AES
